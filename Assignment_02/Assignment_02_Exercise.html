def power_iter(X, num_iter:int):

      v = np.random.randn(X.shape[1])  # Initialize with a random vector with shape (d,)
      one_vec = np.ones(X.shape[0])  # All ones vector with shape (n,)

      mu_row_matrix = np.mean(X, axis=0)  # Returns a 1-row matrix with shape (d, 1)
      mu = np.array(mu_row_matrix).squeeze()  # Convert from a sparse column matrix to a dense 1D array with shape (d,)

      # for _ in range(num_iter):

      #     # Compute the norm of the resulting vector

      #     # Update the singular vector approximation
      #     v = Xv / norm
